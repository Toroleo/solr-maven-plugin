26      /**
27       * Resolve a group of artifacts.
28       *
29       * @param artifacts group of artifacts to be resolved.
30       */
31      public void resolve(List<Artifact> artifacts) {
32          // Brute-force implementation
33          for (Artifact artifact : artifacts) {
34              resolve(artifact);
35          }
36      }

38      /**
39       * Resolve an artifact.
40       *
41       * @param artifact artifact to be resolved.
42       * @return a new <b>instance</b> of resolved artifact.
43       */
44      public Artifact resolve(Artifact artifact) {
45          RepositorySystem repositorySystem = configuration.getRepositorySystem();
46          RepositorySystemSession repositorySession = configuration.getRepositorySession();
47  
48          ArtifactRequest request = new ArtifactRequest();
49          request.setArtifact(artifact);
50          request.setRepositories(configuration.getRemoteRepositories());
51  
52          try {
53              ArtifactResult result =
54                      repositorySystem.resolveArtifact(repositorySession, request);
55              return result.getArtifact();
56          } catch (ArtifactResolutionException e) {
57              throw new RuntimeException(e.getMessage(), e);
58          }
59      }
60  
61      /**
62       * Resolve an artifact by its coordinates.
63       *
64       * @param coordinates coordinates of an artifact to be resolved.
65       * @return resolved artifact.
66       */
67      public Artifact resolve(String coordinates) {
68          return resolve(new DefaultArtifact(coordinates));
